gRPC是什么

gRPC是一种现代化开源的高性能RPC框架，能够运行于任意环境之中。最初由谷歌进行开发。它使用HTTP/2作为传输协议。

使用gRPC， 我们可以一次性的在一个.proto文件中定义服务并使用任何支持它的语言去实现客户端和服务端，反过来，它们可以应用在各种场景中，从Google的服务器到你自己的平板电脑—— gRPC帮你解决了不同语言及环境间通信的复杂性。使用protocol buffers还能获得其他好处，包括高效的序列化，简单的IDL以及容易进行接口更新。总之一句话，使用gRPC能让我们更容易编写跨语言的分布式代码。

IDL（Interface description language）是指接口描述语言，是用来描述软件组件接口的一种计算机语言，是跨平台开发的基础。IDL通过一种中立的方式来描述接口，使得在不同平台上运行的对象和用不同语言编写的程序可以相互通信交流；比如，一个组件用C++写成，另一个组件用Go写成。

普通、一元RPC

普通 rpc，客户端向服务器发送一个请求，然后得到一个响应，就像普通的函数调用一样。

.proto文件中的GetFeature，发送一个坐标，返回坐标的地理名称。

服务端流式RPC

其中客户端向服务器发送请求，并获得一个流来读取一系列消息。客户端从返回的流中读取，直到没有更多的消息。gRPC 保证在单个 RPC 调用中的消息是有序的。

.proto文件中的ListFeatures，发送一个高低坐标构成一片区域，流式返回区域内的地理名称。

客户端流式RPC

其中客户端写入一系列消息并将其发送到服务器，同样使用提供的流。一旦客户端完成了消息的写入，它就等待服务器读取消息并返回响应。同样，gRPC 保证在单个 RPC 调用中对消息进行排序。

.proto文件中的RecordRoute，客户端流式发送坐标点，服务端返回坐标点的总数信息。

双向流式RPC

其中双方使用读写流发送一系列消息。这两个流独立运行，因此客户端和服务器可以按照自己喜欢的顺序读写: 例如，服务器可以等待接收所有客户端消息后再写响应，或者可以交替读取消息然后写入消息，或者其他读写组合。每个流中的消息是有序的。

.proto文件中的RouteChat，客户端流式发送坐标点，服务端流式返回之前经历过的坐标点。

 14def get_feature(feature_db, point):
 15    """Returns Feature at given location or None."""
 16    for feature in feature_db:
 17        if feature.location == point:
 18            return feature
 19    return None
 20
 21
 22def get_distance(start, end):
 23    """Distance between two points."""
 24    coord_factor = 10000000.0
 25    lat_1 = start.latitude / coord_factor
 26    lat_2 = end.latitude / coord_factor
 27    lon_1 = start.longitude / coord_factor
 28    lon_2 = end.longitude / coord_factor
 29    lat_rad_1 = math.radians(lat_1)
 30    lat_rad_2 = math.radians(lat_2)
 31    delta_lat_rad = math.radians(lat_2 - lat_1)
 32    delta_lon_rad = math.radians(lon_2 - lon_1)
 33
 34    # Formula is based on http://mathforum.org/library/drmath/view/51879.html
 35    a = (pow(math.sin(delta_lat_rad / 2), 2) +
 36         (math.cos(lat_rad_1) * math.cos(lat_rad_2) *
 37          pow(math.sin(delta_lon_rad / 2), 2)))
 38    c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a))
 39    R = 6371000
 40    # metres
 41    return R * c
 42

 44class RouteGuideServicer(route_guide_pb2_grpc.RouteGuideServicer):
 45    """Provides methods that implement functionality of route guide server."""
 46
 47    def __init__(self):
 48        self.db = route_guide_resources.read_route_guide_database()
 49
 50    def GetFeature(self, request, context):
 51        feature = get_feature(self.db, request)
 52        if feature is None:
 53            return route_guide_pb2.Feature(name="", location=request)
 54        else:
 55            return feature
 56
 57    def ListFeatures(self, request, context):
 58        left = min(request.lo.longitude, request.hi.longitude)
 59        right = max(request.lo.longitude, request.hi.longitude)
 60        top = max(request.lo.latitude, request.hi.latitude)
 61        bottom = min(request.lo.latitude, request.hi.latitude)
 62        for feature in self.db:
 63            if (feature.location.longitude >= left and
 64                    feature.location.longitude <= right and
 65                    feature.location.latitude >= bottom and
 66                    feature.location.latitude <= top):
 67                yield feature
 68
 69    def RecordRoute(self, request_iterator, context):
 70        point_count = 0
 71        feature_count = 0
 72        distance = 0.0
 73        prev_point = None
 74
 75        start_time = time.time()
 76        for point in request_iterator:
 77            point_count += 1
 78            if get_feature(self.db, point):
 79                feature_count += 1
 80            if prev_point:
 81                distance += get_distance(prev_point, point)
 82            prev_point = point
 83
 84        elapsed_time = time.time() - start_time
 85        return route_guide_pb2.RouteSummary(point_count=point_count,
 86                                            feature_count=feature_count,
 87                                            distance=int(distance),
 88                                            elapsed_time=int(elapsed_time))
 89
 90    def RouteChat(self, request_iterator, context):
 91        prev_notes = []
 92        for new_note in request_iterator:
 93            for prev_note in prev_notes:
 94                if prev_note.location == new_note.location:
 95                    yield prev_note
 96            prev_notes.append(new_note)
 97
 98
 99def serve():
100    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
101    route_guide_pb2_grpc.add_RouteGuideServicer_to_server(
102        RouteGuideServicer(), server)
103    server.add_insecure_port('[::]:50051')
104    server.start()
105    server.wait_for_termination()

109    logging.basicConfig()
110    serve()

 69// RecordRoute records a route composited of a sequence of points.
 70//
 71// It gets a stream of points, and responds with statistics about the "trip":
 72// number of points,  number of known features visited, total distance traveled, and
 73// total time spent.
 74func (s *routeGuideServer) RecordRoute(stream pb.RouteGuide_RecordRouteServer) error {
 75	var pointCount, featureCount, distance int32
 76	var lastPoint *pb.Point
 77	startTime := time.Now()
 78	for {
 79		point, err := stream.Recv()
 80		if err == io.EOF {
 81			endTime := time.Now()
 82			return stream.SendAndClose(&pb.RouteSummary{
 83				PointCount:   pointCount,
 84				FeatureCount: featureCount,
 85				Distance:     distance,
 86				ElapsedTime:  int32(endTime.Sub(startTime).Seconds()),
 87			})
 88		}
 89		if err != nil {
 90			return err
 91		}
 92		pointCount++
 93		for _, feature := range s.savedFeatures {
 94			if proto.Equal(feature.Location, point) {
 95				featureCount++
 96			}
 97		}
 98		if lastPoint != nil {
 99			distance += calcDistance(lastPoint, point)
100		}
101		lastPoint = point
102	}

105// RouteChat receives a stream of message/location pairs, and responds with a stream of all
106// previous messages at each of those locations.
107func (s *routeGuideServer) RouteChat(stream pb.RouteGuide_RouteChatServer) error {
108	for {
109		in, err := stream.Recv()
110		if err == io.EOF {
111			return nil
112		}
113		if err != nil {
114			return err
115		}
116		key := serialize(in.Location)
117
118		s.mu.Lock()
119		s.routeNotes[key] = append(s.routeNotes[key], in)
120		// Note: this copy prevents blocking other clients while serving this one.
121		// We don't need to do a deep copy, because elements in the slice are
122		// insert-only and never modified.
123		rn := make([]*pb.RouteNote, len(s.routeNotes[key]))
124		copy(rn, s.routeNotes[key])
125		s.mu.Unlock()
126
127		for _, note := range rn {
128			if err := stream.Send(note); err != nil {
129				return err
130			}
131		}
132	}

152func toRadians(num float64) float64 {
153	return num * math.Pi / float64(180)

156// calcDistance calculates the distance between two points using the "haversine" formula.
157// The formula is based on http://mathforum.org/library/drmath/view/51879.html.
158func calcDistance(p1 *pb.Point, p2 *pb.Point) int32 {
159	const CordFactor float64 = 1e7
160	const R = float64(6371000) // earth radius in metres
161	lat1 := toRadians(float64(p1.Latitude) / CordFactor)
162	lat2 := toRadians(float64(p2.Latitude) / CordFactor)
163	lng1 := toRadians(float64(p1.Longitude) / CordFactor)
164	lng2 := toRadians(float64(p2.Longitude) / CordFactor)
165	dlat := lat2 - lat1
166	dlng := lng2 - lng1
167
168	a := math.Sin(dlat/2)*math.Sin(dlat/2) +
169		math.Cos(lat1)*math.Cos(lat2)*
170			math.Sin(dlng/2)*math.Sin(dlng/2)
171	c := 2 * math.Atan2(math.Sqrt(a), math.Sqrt(1-a))
172
173	distance := R * c
174	return int32(distance)
175}

177func inRange(point *pb.Point, rect *pb.Rectangle) bool {
178	left := math.Min(float64(rect.Lo.Longitude), float64(rect.Hi.Longitude))
179	right := math.Max(float64(rect.Lo.Longitude), float64(rect.Hi.Longitude))
180	top := math.Max(float64(rect.Lo.Latitude), float64(rect.Hi.Latitude))
181	bottom := math.Min(float64(rect.Lo.Latitude), float64(rect.Hi.Latitude))
182
183	if float64(point.Longitude) >= left &&
184		float64(point.Longitude) <= right &&
185		float64(point.Latitude) >= bottom &&
186		float64(point.Latitude) <= top {
187		return true
188	}
189	return false
190}

211			*certFile = data.Path("x509/server_cert.pem")
212		}
213		if *keyFile == "" {
214			*keyFile = data.Path("x509/server_key.pem")
215		}
216		creds, err := credentials.NewServerTLSFromFile(*certFile, *keyFile)
217		if err != nil {
218			log.Fatalf("Failed to generate credentials %v", err)
219		}
220		opts = []grpc.ServerOption{grpc.Creds(creds)}
221	}
222	grpcServer := grpc.NewServer(opts...)
223	pb.RegisterRouteGuideServer(grpcServer, newServer())
224	grpcServer.Serve(lis)
225}

 63// runRecordRoute sends a sequence of points to server and expects to get a RouteSummary from server.
 64func runRecordRoute(client pb.RouteGuideClient) {
 65	// Create a random number of random points
 66	r := rand.New(rand.NewSource(time.Now().UnixNano()))
 67	pointCount := int(r.Int31n(100)) + 2 // Traverse at least two points
 68	var points []*pb.Point
 69	for i := 0; i < pointCount; i++ {
 70		points = append(points, randomPoint(r))
 71	}
 72	log.Printf("Traversing %d points.", len(points))
 73	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
 74	defer cancel()
 75	stream, err := client.RecordRoute(ctx)
 76	if err != nil {
 77		log.Fatalf("client.RecordRoute failed: %v", err)
 78	}
 79	for _, point := range points {
 80		if err := stream.Send(point); err != nil {
 81			log.Fatalf("client.RecordRoute: stream.Send(%v) failed: %v", point, err)
 82		}
 83	}
 84	reply, err := stream.CloseAndRecv()
 85	if err != nil {
 86		log.Fatalf("client.RecordRoute failed: %v", err)
 87	}
 88	log.Printf("Route summary: %v", reply)
 89}

 91// runRouteChat receives a sequence of route notes, while sending notes for various locations.
 92func runRouteChat(client pb.RouteGuideClient) {
 93	notes := []*pb.RouteNote{
 94		{Location: &pb.Point{Latitude: 0, Longitude: 1}, Message: "First message"},
 95		{Location: &pb.Point{Latitude: 0, Longitude: 2}, Message: "Second message"},
 96		{Location: &pb.Point{Latitude: 0, Longitude: 3}, Message: "Third message"},
 97		{Location: &pb.Point{Latitude: 0, Longitude: 1}, Message: "Fourth message"},
 98		{Location: &pb.Point{Latitude: 0, Longitude: 2}, Message: "Fifth message"},
 99		{Location: &pb.Point{Latitude: 0, Longitude: 3}, Message: "Sixth message"},
100	}
101	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
102	defer cancel()
103	stream, err := client.RouteChat(ctx)
104	if err != nil {
105		log.Fatalf("client.RouteChat failed: %v", err)
106	}
107	waitc := make(chan struct{})
108	go func() {
109		for {
110			in, err := stream.Recv()
111			if err == io.EOF {
112				// read done.
113				close(waitc)
114				return
115			}
116			if err != nil {
117				log.Fatalf("client.RouteChat failed: %v", err)
118			}
119			log.Printf("Got message %s at point(%d, %d)", in.Message, in.Location.Latitude, in.Location.Longitude)
120		}
121	}()
122	for _, note := range notes {
123		if err := stream.Send(note); err != nil {
124			log.Fatalf("client.RouteChat: stream.Send(%v) failed: %v", note, err)
125		}
126	}
127	stream.CloseSend()
128	<-waitc
129}

131func randomPoint(r *rand.Rand) *pb.Point {
132	lat := (r.Int31n(180) - 90) * 1e7
133	long := (r.Int31n(360) - 180) * 1e7
134	return &pb.Point{Latitude: lat, Longitude: long}
